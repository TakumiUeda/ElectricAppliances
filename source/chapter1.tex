\chapter{直流機}
\section{回転機械の持つエネルギー}
\subsection{慣性質量と慣性モーメント}
\begin{align}
F=m \alpha = m \drv{v}{t} = m \sdrv{x}{t} \mathrm{[N]}
\end{align}

\begin{align}
\tau =\int F dx =\int m \drv{v}{t} dx =\int m v dv = \frac{1}{2} m v^{2} \mathrm{[J]}
\end{align}

\begin{align}
v =\drv{x}{t} = r \drv{\theta}{t} = r \omega \mathrm{[m/s]}
\end{align}

\begin{align}
F = m \drv{t}{v} = m r \drv{\omega}{t} \mathrm{[N]}
\end{align}

\begin{align}
T = Fr = mr^{2} \drv{\omega}{t} = I \drv{\omega}{t} = I \sdrv{\theta}{t} \mathrm{[N]}
\end{align}
Iは慣性モーメント\\
\subsection{慣性モーメントの定義}
トルクはそれぞれの物体が回転するために必要なトルクの合計になるので
\begin{align}
T = \sum_{i=1}^{n} F_{i}r_{i} = \sum_{i=1}^{n} m_{i} r^{2}_{i} \drv{\omega}{t} = I \drv{\omega}{t}
\end{align}
となり，慣性モーメント$I$は
\begin{align}
I = \sum_{i=1}^{n} m_{i} r^{2}_{i}
\end{align}

\begin{align}
I = \sum_{i=1}^{\infty} m_{i} r^{2}_{i}
\end{align}

\begin{align}
I = \int r^{2} dm
\end{align}
